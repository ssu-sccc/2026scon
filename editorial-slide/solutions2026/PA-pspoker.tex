\begin{frame}{\probno{}. \probtitle{}}
    \begin{itemize}
        \item 각 사람의 값은 $C_i = \frac{B_i}{A_i}$입니다.
        \item $\frac{B_i}{A_i}$가 가장 큰 사람을 구하면 됩니다.
        \item 실수 비교 대신
        \[
            B_i A_j < B_j A_i
        \]
        로 비교합니다.
        \item 같다면 번호가 큰 사람을 고릅니다.
    \end{itemize}
\end{frame}
